BRT_manage_area_quotas = {
	BRT_low_area_quota = {
		icon = generic_decision

		target_trigger = {
			FROM = {
				is_subject_of = BRT
				has_country_flag = hbe_administrative_area
			}
		}

		available = {
		}

		visible = {
			FROM = {
				OR = {
					has_idea = quota_level_2
					has_idea = quota_level_3
				}
			}
		}

		cost = 0
		fire_only_once = no

		complete_effect = {
			IF = {
				limit = {
					FROM = {
						has_idea = quota_level_2
					}
				}

				FROM = {
					swap_ideas = {
						remove_idea = quota_level_2
						add_idea = quota_level_1
					}
				}
			}
			else_if = {
				limit = {
					FROM = {
						has_idea = quota_level_3
					}
				}

				FROM = {
					swap_ideas = {
						remove_idea = quota_level_3
						add_idea = quota_level_1
					}
				}
			}
		}
	}
