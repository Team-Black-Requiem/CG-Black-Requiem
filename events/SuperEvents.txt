	hidden = yes

	major = yes
	timeout_days = 1

			set_country_flag = SUPER_JAP_invasion							#country flag that carries title localization - should always be prefixed with SUPER_

			set_variable = {BR_Super_Event_Var = token:SUPER_JAP_invasion}  #BR_Super_Event_Var stores the token ID of the title country flag so we can do some voodoo
		}
}
